<script>

// Google Sheet CSV 網址

const sheetURL="https://docs.google.com/spreadsheets/d/1zYKdMH5pFt9d7X31Vp6hnEm8wnC2JzbKTfdkVoV1tPA/export?format=csv";

function extractDriveID(url){

const match=url.match(/[-\w]{25,}/);

return match?match[0]:url;

}

function driveImage(url){

const id=extractDriveID(url);

return id;

}

function driveDownload(url){

const id=extractDriveID(url);

return id;

}

fetch(sheetURL)

.then(res=>res.text())

.then(text=>{

const rows=text.split(/\r?\n/);

const container=document.getElementById("timeline");

for(let i=1;i<rows.length;i++){

const row=rows[i];

if(!row.trim()) continue;

const cols=row.split(/,(?=(?:(?:[^"]*"){2})*[^"]*$)/);

const date=cols[0];
const title=cols[1];
const content=cols[2];

let galleryHTML="";

if(cols[3]){

const imgs=cols[3].split(",");

galleryHTML="<div class='gallery'>";

imgs.forEach(img=>{

const src=driveImage(img);

galleryHTML+=`<img src="${src}">`;

});

galleryHTML+="</div>";

}

let pdfHTML="";

if(cols[4]){

const link=driveDownload(cols[4]);

pdfHTML=`

<a href="${link}" class="download-btn">

<span class="material-symbols-outlined">download</span>

下載文件

</a>

`;

}

container.innerHTML+=`

<div class="timeline-item">

<details>

<summary>

<div class="timeline-header">

<div class="timeline-dot"></div>

<div class="timeline-title">

${date} ${title}

</div>

<span class="material-symbols-outlined icon">expand_more</span>

</div>

</summary>

<div class="content">

${content}

${galleryHTML}

${pdfHTML}

</div>

</details>

</div>

`;

}

const items=document.querySelectorAll(".timeline details");

items.forEach(el=>{

el.addEventListener("toggle",function(){

if(this.open){

items.forEach(o=>{
if(o!==this){
o.removeAttribute("open");
}
});

}

});

});

const images=document.querySelectorAll(".gallery img");

const lightbox=document.getElementById("lightbox");

const lightboxImg=document.getElementById("lightbox-img");

const close=document.getElementById("close");

images.forEach(img=>{

img.onclick=()=>{

lightbox.style.display="flex";

lightboxImg.src=img.src;

};

});

close.onclick=()=>{

lightbox.style.display="none";

};

});

</script>
